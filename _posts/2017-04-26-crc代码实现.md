知识点：
* 十六进制转化为十进制，再转化为二进制
* 溢出错误
* CRC涉及到的二进制除法，异或减

过程：
1. 发送端把冗余信息附在要发送的信息后
2. 对整个发送信息进行加密处理（输入密钥）
3. 发送
4. 接收方收到加密的信息先进行解密（输入发送方的密钥）
5. 进行完整性认证（余数为0）

## // CR4加密 CR4.h
```c

#include <iostream>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
using namespace std;
void init();
void Encode(char txt[]);
char key[257];
char T[257];
int S[257];

void init(){
  int len = strlen(key);
  for(int i=0; i<256; i++){
    S[i] = i;/*初始化S盒*/ 
    T[i] = key[i % len];/*密钥填充256数组*/  
  }
  //用T产生S的初始置换
  int j = 0;
  for(int i=0; i<256; i++){
    j = (j + S[i] + T[i]) % 256;
    S[i] = S[i] ^ S[j];
    S[j] = S[i] ^ S[j];
    S[i] = S[i] ^ S[j];
  }
}

void Encode(char txt[]){
  cout<<endl<<"input key: "<<endl;
  scanf("%s", key);
  init();
  int i, j=0;
  for(i=0; txt[i]!=0; i++){
    i = (i+1) % 256;
    j = (j+S[i]) % 256;
    S[i] = S[i] ^ S[j];
    S[j] = S[i] ^ S[j];
    S[i] = S[i] ^ S[j];
    int t = (S[i] + S[j]) % 256;
    int k = S[t]; 
    txt[i] ^= k;
  }
  cout<<"result:  ";
  cout<<txt<<endl<<endl;
}
```

## CRC循环冗余校验法 CRC.cpp
```c
#include <iostream>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include "CR4.h"
using namespace std;

/*
numerator 被除数 
denominator 除数 
首先是校验码的长度order，然后是生成多项式polynom，
其次是初始值initial value，
最后是异或（Xor）值。此外还有两个选项，一个是“reverse data bytes”反转数据，
另一个是“reverse CRC result result before Final XOR”在最后进行异或运算之前先反转CRC的结果值。
   CRC-4       x4+x+1               
   CRC-8       x8+x5+x4+1                                
   CRC-8       x8+x2+x1+1                               
   CRC-8       x8+x6+x4+x3+x2+x1        
   CRC-12      x12+x11+x3+x+1          
   CRC-16      x16+x15+x2+1            
   CRC16-CCITT x16+x12+x5+1  
   任意指定码多项式二进制串（16进制表示）
   对学号（重复3次）进行完整性校验后加密，给出结果
 */

char numerator[150];
char numeratorTemp[150];
char denominator[150];
char FCS[150];
int FCSLen;
int numeratorLen;
int denominatorLen;

//十六进制转为二进制
void _16to_2(char input[]){
  char ans[150];
  int len = strlen(input);
  cout<<"input len: "<<len<<"  input:  "<<input<<endl;
  long long _10 = 0,x,weight=1;
  for(int i=len-1; i>=0; i--){
    
    if(input[i]>='0'&&input[i]<='9'){
      x = input[i] - '0';
    } else if(input[i]>='a'&&input[i]<='z'){
      x = input[i] - 'a' + 10;
    } else if(input[i]>='A'&&input[i]<='Z'){
      x = input[i] - 'A' + 10;
    }
    _10 += x * weight;
    weight *= 16;
  }
  cout<<"_10: "<<_10<<endl;
  //把十进制转化为二进制
  len *= 4;
  ans[len] = 0;
  for(int i=len-1; i>=0; i--){
    int x = _10 % 2;
    ans[i] = x + '0';
    _10 /= 2;
  }
  cout<<"input len: "<<len<<"  input:  "<<ans<<endl;
  while(ans[0]=='0'){
    int i;
    for(i=1; i<strlen(ans); i++){
      ans[i-1] = ans[i];
    }
    ans[i-1] = 0;
  }
  strcpy(input, ans);

}

void XOR(char beijianshu[], char jianshu[]){
  for(int i=0; i<denominatorLen; i++){
    beijianshu[i] = ((beijianshu[i]-'0') ^ (jianshu[i] - '0')) + '0';
  }
}

void CRC(){
  numeratorLen = strlen(numerator);
  denominatorLen = strlen(denominator);
  FCSLen = denominatorLen - 1;
  FCS[FCSLen] = 0;
  //补零
  for(int i=0; i<FCSLen; i++){
    strcat(numerator, "0");
  }
  int nowNumeratorLen = strlen(numerator);

  //除法
  int index = 0;//指针指向被除数正在运算的位
  char ans[150];//每次的结果
  for(; index<denominatorLen; index++){//初始化
    ans[index] = numerator[index];
  }
  ans[index--] = 0;
  while(1){
    //统计ans前面0的个数，去1个0，index后移1位补到ans，直到前面0去完了或者index指向最后一个
    while(ans[0]=='0'){
      int i;
      for(i=1; i<denominatorLen; i++){
        ans[i-1] = ans[i];
      }
      index++;
      ans[i-1] = numerator[index];
      if(index == nowNumeratorLen-1){
        ans[i] = 0;
        break;
      }
    }
    if(index == nowNumeratorLen-1){
      int i;
      for(i=1; i<denominatorLen; i++){
        ans[i-1] = ans[i];
      }
      ans[i-1] = 0;
      break;
    }
    XOR(ans, denominator);
  }
  strcpy(FCS, ans);
  strcpy(numerator, numeratorTemp);//去掉后面的0
  //此时为真正发送的
  strcat(numerator, FCS);//加上fcs
  cout<<"[";for(int i=1; i<numeratorLen-1; i++){cout<<" ";}
  cout<<"][";for(int i=1; i<FCSLen-1; i++){cout<<" ";}cout<<"]"<<endl;
  cout<<numerator<<endl;
  cout<<"FCS: "<<FCS<<endl;
}
```
